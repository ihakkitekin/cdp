function ChangeDirectoryPartially() {
    $currentDir = Get-Location

    Foreach($i in $args) {
        $result = Get-ChildItem $currentDir *$i* -Depth 0 -Recurse -Directory | Select-Object -First 1
	If ($result.length -gt 0){
            $currentDir = $result.FullName
	}  
	Else {
	    Return echo "There is no match for `"$i`" exiting search."
        }       
    }

    cd $currentDir
}

Set-Alias cdp ChangeDirectoryPartially
